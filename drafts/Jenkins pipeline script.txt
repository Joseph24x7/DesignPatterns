pipeline {
    agent any
    tools{
        maven 'MAVEN_HOME'
    }
    stages{
        stage('Build Maven'){
            steps{
                checkout([$class: 'GitSCM', branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/Joseph24x7/BookMyGift.git']]])
                bat 'mvn clean install'
            }
        }
        stage('Code Scan'){
            steps{
                checkout([$class: 'GitSCM', branches: [[name: '*/main']], extensions: [], userRemoteConfigs: [[url: 'https://github.com/Joseph24x7/BookMyGift.git']]])
                bat 'mvn clean verify sonar:sonar -Dsonar.projectKey=BookMyGift -Dsonar.login=admin -Dsonar.password=Password'
            }
        }
        stage('Build docker image'){
            steps{
                script{
                    bat 'docker build -t joseph24x7/bookmyticket .'
                }
            }
        }
        stage('Push image to Hub'){
            steps{
                script{
                   bat 'docker login -u joseph24x7 -p Password'
                   bat 'docker push joseph24x7/bookmyticket'
                }
            }
        }
    }
}
